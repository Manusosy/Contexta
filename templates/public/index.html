contexta.js
import Contexta from '@contexta/sdk';

// Instantiate once, use everywhere
const ctx = new Contexta({
  apiKey: process.env.CONTEXTA_KEY,
  region: 'af-east-1'
});

// Activate LexKenya research agent
const result = await ctx.agents.activate({
  agent:  'lexkenya-research',
  query:  'right to fair trial',
  corpus: 'constitution-ke-2010'
});

// Returns structured citations
console.log(result.citations);
